174      public void setData(String url) {
175          Model_simple_staff_list model = new Model_simple_staff_list(getStringFromInput(binding.setName, "text"),
176                  getStringFromInput(binding.setPost, "text"),
177                  url,
178                  getStringFromInput(binding.setSection, "text"),
179                  getStringFromInput(binding.setHomeDistrict, "text"),
180                  getStringFromInput(binding.setEducation, "text"),
181                  getStringFromInput(binding.setGovtJobJoiningDate, "text"),
182                  getStringFromInput(binding.setCurrentPositionJoinningDate, "text"),
183                  getStringFromInput(binding.setCurrentInstituteJoinningDate, "text"),
184                  getStringFromInput(binding.setMobile, "text"),
185                  getStringFromInput(binding.setEmail, "email"),
186                  getIntegerFromInput(binding.setPayority, "Sort Number"));
187  
188          db.collection(collection).document(getStringFromInput(binding.setName, "text")).set(model, SetOptions.merge()).addOnSuccessListener(new OnSuccessListener<Void>() {
189                      @Override
190                      public void onSuccess(Void unused) {
191                          dialog.dismiss();
192                          Toast.makeText(setPeople.this, "successfully submitted", Toast.LENGTH_SHORT).show();

202                      }
