41       @Override
42       public void onBindViewHolder(@NonNull viewholder viewholder, int i) {
43           Model_simple_staff_list model = list.get(i);
44           viewholder.Name.setText(list.get(i).getSimple_staff_list_Name());
45           viewholder.Post.setText(list.get(i).getSimple_staff_list_post());
46           Picasso.get().load(list.get(i).getImagelink()).placeholder(R.drawable.ic_baseline_account_circle_24).into(viewholder.Photo);
47           viewholder.itemView.setOnLongClickListener(new View.OnLongClickListener() {
48               @Override
49               public boolean onLongClick(View view) {
50                   setIntentForSetPeaple(context, setPeople.class, "officeStaffInfo", model.getSimple_staff_list_Name());
51                   return false;
52               }
53           });
54           viewholder.itemView.setOnClickListener(new View.OnClickListener() {
55               @Override
56               public void onClick(View view) {
57                   Intent intent = new Intent(context, staff_list_in_details.class);
58                   intent.putExtra("name", model.getSimple_staff_list_Name());
59                   intent.putExtra("post", model.getSimple_staff_list_post());
60                   intent.putExtra("section", model.getSection());
61                   intent.putExtra("homeDistrict", model.getHomeDistrict());
62                   intent.putExtra("education", model.getEducation());
63                   intent.putExtra("govtJobJoiningDate", model.getGovtJobJoiningDate());
64                   intent.putExtra("currentPositionJoinningDate", model.getCurrentPositionJoinningDate());
65                   intent.putExtra("currentInstituteJoinningDate", model.getCurrentInstituteJoinningDate());
66                   intent.putExtra("mobile", model.getMobile());
67                   intent.putExtra("email", model.getEmail());
68                   intent.putExtra("photo", model.getSimple_staff_list_Image());
69                   intent.putExtra("photoLink", model.getImagelink());
70                   context.startActivity(intent);
71   
72               }
73           });
